冒泡排序

原理: 相邻两个数据比较,每次找到小的放前面
原始数据:{1,3,4,2,6,5,7,9,8}
第一轮:{1,3,2,4,5,6,7,8,9}
第二轮:{1,2,343,5,6,7,8,9}
…
代码:

public void insertSort(int[] a){
    int N=a.length;
    //第一个不动
    //第二次开始和前面的比较,遇到比自己小的就停下,插入,后面的数值后移,否则就保持放到最后
    for(int i=1;i<N;i++){
        int temp=a[i];
        for (int j=i-1;j>=0;j--){
            if(j==i-1&&temp>a[j]){
                break;
            }else if (temp>a[j]){
                a[j+1]=temp;
                break;
            }
            a[j+1]=a[j];
        }
    }
}

public void shellSort(int [] a){
int N=a.length;
int dk=N/2;
while(dk>=1){
insertSort(a,dk);
dk=dk/2;
}
}

for(int i=0;i<N;i=i+dk){
int temp=a[i];
for (int j=i-dk;j>=0;j=j-dk){
if(j==i-dk&&temp>a[j]){
break;

}
}

public int [] merge(int []a1,int[] a2){
    //两个数组是已经排好序的
    int n1=a1.length;
    int n2=a2.length;
    int [] a=new int[n1+n2];
    int i=0,j=0,k=0;
    while (i<n1&&j<n2){
        if(a1[i]<a2[j]){
            a[k++]=a1[i++];
        }else {
            a[k++]=a2[j++];
        }
    }
    while (i<n1){
        a[k++]=a1[i++];
    }
    while (j<n2){
        a[k++]=a2[j++];
    }
    return a;
}
